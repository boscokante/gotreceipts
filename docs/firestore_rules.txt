rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Rule 1: This is for the mobile app user. It allows them to read and write
    // any documents inside their own user folder (e.g., /users/USER_ID/...)
    match /users/{userId}/{document=**} {
      allow read, write: if request.auth.uid == userId;
    }

    // Rule 2: This is for the web dashboard. It allows any signed-in user
    // (like a bookkeeper) to perform a 'list' (or query) across the 'receipts'
    // collection group. This is what allows the query to succeed.
    match /{path=**}/receipts/{receiptId} {
      allow list: if request.auth != null;
    }
  }
}
