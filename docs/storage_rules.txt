rules_version = '2';

// This is the configuration for Firebase Cloud Storage.
// It allows users to read and write files only if they are signed in to the app.
// This is a secure default for applications like ours.

service firebase.storage {
  match /b/{bucket}/o {
    match /{allPaths=**} {
      allow read, write: if request.auth != null;
    }
  }
}
